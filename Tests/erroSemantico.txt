program EXEMPLO_WHILE;
var
 A, R, I, Z : integer;
 B: boolean;
begin
 while (I <= 5) do
 begin
 write('Entre um valor para A: ');
 readln(A);
 write('Entre um valor para B: ');
 readln(B);
 writeln;
 R := A + B;
 writeln('O resultado corresponde a: ',R);
 writeln;
 I := I + 1;
 end;
end.
